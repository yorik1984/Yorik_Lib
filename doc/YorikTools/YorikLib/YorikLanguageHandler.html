def initialize(plugin_id, resources_dir, default_localization)
  @plugin_id = plugin_id
  @resources_dir = resources_dir
  @default_localization = default_localization
  @default_lang_name_ = "English"
  @localization_file_type = ".strings"
  @lang_ini_file = "lang.ini"
  @lang_dir_ini_file = "lang_dir.ini"
  @strings = Hash.new { |hash, key| key }
  check_default_ini_files
  @lang_ini_error_msg      = "Error to initialize #{@lang_file_path}"
  @lang_dir_ini_error_msg  = "Error to initialize #{@lang_dir_file_path}"
  @dir_list_error_msg      =
      "Error in file #{@lang_dir_file_path} with directories list localization strings.\n"
  @default_using_error_msg =
      "It will be available choose only English(default) localization.\n" +
      "Please, see more in Tools > Yorik Lib > Help"
  parse
end

def check_default_ini_files
  @lang_file_path = File.join(@resources_dir, @lang_ini_file)
  unless File.exist?(@lang_file_path)
    make_default_lang_files(@lang_file_path)
    @lang_ini_error = true
  end
  @lang_dir_file_path = File.join(@resources_dir, @lang_dir_ini_file)
  unless File.exist?(@lang_dir_file_path)
    @lang_dir_ini_error = true
    make_default_lang_files(@lang_dir_file_path, @default_lang_name + " ")
  end
end

def dir_list(dir)
  entries = Dir.entries(dir)
  entries.delete_if { |elem| elem == "." || elem == ".." }
  sud_dir_list = []
  entries.each { |entrie|
    sud_dir_list.push entrie if Dir.exist?(File.join(dir, entrie)) }
  sud_dir_list.sort!
end

def find_strings_file
  localization = @default_localization
  File.open(@lang_file_path).each { |line| localization = line.chomp }
  strings_file = strings_file_path(localization)
  return strings_file if File.exist?(strings_file)
  strings_file_path(@default_localization)
end

def localization_mbox(inputbox_window_name)
  localization_dir_list = make_localization_dir_list
  UI.messagebox(@lang_ini_error_msg) if @lang_ini_error
  UI.messagebox(@lang_dir_ini_error_msg) if @lang_dir_ini_error
  UI.messagebox(@dir_list_error_msg) if @dir_list_error
  UI.messagebox(@default_using_error_msg) if @lang_dir_ini_error ||
      @lang_ini_error || @dir_list_error
  @lang_ini_error     = false
  @lang_dir_ini_error = false
  @dir_list_error     = false
  list_keys = localization_dir_list.keys
  prompts = [inputbox_window_name]
  defaults = []
  defaults[0] = localization_dir_list[list_keys[0]]
  list = []
  list[0] = ""
  localization_dir_list.each_value { |value|
    list[0] = list[0] + value.to_s + "|" }
  inputbox = UI.inputbox(prompts, defaults, list, inputbox_window_name)
  current_localization = localization_dir_list.key(inputbox[0]).to_s
  write_localization_to_file(current_localization)
end

def make_localization_dir_list
  result = []
  localization_list = read_localization_list_file
  key_list_of_dir = localization_list.keys
  key_list_of_dir.delete_if { |elem| elem.nil?}
  diff = key_list_of_dir.size - localization_list.size
  unless dir_list(@resources_dir) == key_list_of_dir.sort! &&
      diff == 0
    key = (@default_localization).to_sym
    @dir_list_error = true
    return localization_list = Hash[key, @default_lang_name]
  end
  localization_list
end

def read_localization_list_file
  localization_list = {}
  File.open(@lang_dir_file_path, 'r:BOM|UTF-8').each do |line|
    scan = line.scan(/[[:word:]]+/)
    next if scan.size == 0
    language = scan[0]
    localization = scan[1]
    localization_list[localization] = language
  end
  localization_list
end
